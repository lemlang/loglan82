cmake_minimum_required(VERSION 2.8)
project(vlpc)
set(CMAKE_MODULE_PATH "${PROJECT_SOURCE_DIR}")  
find_package(wxWidgets 3.0 COMPONENTS base REQUIRED)
OPTION( WXWINDOWS_USE_UNICODE "Use Unicode versions of wxWindows libraries" ON)
include( "${wxWidgets_USE_FILE}" )
set(SRCS main.cpp Compiller.cpp IncludeFile.cpp)
#if(WIN32)
#  set(SRCS ${SRCS} minimal.rc)
#endif(WIN32)
add_executable(vlpc WIN32 ${SRCS})
target_link_libraries(vlpc ${wxWidgets_LIBRARIES})
