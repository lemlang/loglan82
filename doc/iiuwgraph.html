unit INKEY:function:integer; This function checks for pressed key but is not waiting for an input. The key code is returned if the key was pressed and 0 if not. Special keys return the following codes: 
F1 = -10   F9 = -18        Right = -26
F2 = -11   F10 = -19       Up = -27
F3 = -12   Insert = -20    Down = -28
F4 = -13   Home = -21      Tab = 9
F5 = -14   End = -22       Esc = 27
F6 = -15   PgUp = -23      Backspace = 8
F7 = -16   PgDown = -24    Del = 117
F8 = -17   Left = -25
